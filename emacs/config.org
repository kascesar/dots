#+BEGIN_SRC emacs-lisp
(defvar elpaca-installer-version 0.12)
(defvar elpaca-directory (expand-file-name "elpaca/" user-emacs-directory))
(defvar elpaca-builds-directory (expand-file-name "builds/" elpaca-directory))
(defvar elpaca-sources-directory (expand-file-name "sources/" elpaca-directory))
(defvar elpaca-order '(elpaca :repo "https://github.com/progfolio/elpaca.git"
                              :ref nil :depth 1 :inherit ignore
                              :files (:defaults "elpaca-test.el" (:exclude "extensions"))
                              :build (:not elpaca-activate)))
(let* ((repo  (expand-file-name "elpaca/" elpaca-sources-directory))
       (build (expand-file-name "elpaca/" elpaca-builds-directory))
       (order (cdr elpaca-order))
       (default-directory repo))
  (add-to-list 'load-path (if (file-exists-p build) build repo))
  (unless (file-exists-p repo)
    (make-directory repo t)
    (when (<= emacs-major-version 28) (require 'subr-x))
    (condition-case-unless-debug err
        (if-let* ((buffer (pop-to-buffer-same-window "*elpaca-bootstrap*"))
                  ((zerop (apply #'call-process `("git" nil ,buffer t "clone"
                                                  ,@(when-let* ((depth (plist-get order :depth)))
                                                      (list (format "--depth=%d" depth) "--no-single-branch"))
                                                  ,(plist-get order :repo) ,repo))))
                  ((zerop (call-process "git" nil buffer t "checkout"
                                        (or (plist-get order :ref) "--"))))
                  (emacs (concat invocation-directory invocation-name))
                  ((zerop (call-process emacs nil buffer nil "-Q" "-L" "." "--batch"
                                        "--eval" "(byte-recompile-directory \".\" 0 'force)")))
                  ((require 'elpaca))
                  ((elpaca-generate-autoloads "elpaca" repo)))
            (progn (message "%s" (buffer-string)) (kill-buffer buffer))
          (error "%s" (with-current-buffer buffer (buffer-string))))
      ((error) (warn "%s" err) (delete-directory repo 'recursive))))
  (unless (require 'elpaca-autoloads nil t)
    (require 'elpaca)
    (elpaca-generate-autoloads "elpaca" repo)
    (let ((load-source-file-function nil)) (load "./elpaca-autoloads"))))
(add-hook 'after-init-hook #'elpaca-process-queues)
(elpaca `(,@elpaca-order))
(elpaca elpaca-use-package
  (elpaca-use-package-mode))

** modeline
Mode line propia, portada de la de [[https://github.com/protesilaos/dotfiles][Protesilaos]] (=prot-modeline.el=). La idea
central es que casi todos los segmentos se dibujan *solo en la ventana
enfocada* (=mode-line-window-selected-p=), así no se repiten los indicadores
globales en cada split.

Adaptaciones respecto al original: sin iconos, ~flycheck~ en lugar de
~flymake~, ~ef-themes~ en lugar de ~modus-themes~, y sin ~eglot~ /
~notmuch~ / ~spacious-padding~.

Nota: este formato no incluye ~mode-line-modes~, así que la lista de minor
modes desaparece a propósito.

*** Helpers y faces
#+begin_src emacs-lisp
  (defcustom my/modeline-string-truncate-length 9
    "Largo a partir del cual se truncan las cadenas de la mode line."
    :type 'natnum
    :group 'mode-line)

  (defun my/modeline--window-narrow-p ()
    "Non-nil si la ventana es estrecha (cf. `split-width-threshold')."
    (and (numberp split-width-threshold)
         (< (window-total-width) split-width-threshold)))

  (defun my/modeline--string-truncate-p (str)
    "Non-nil si STR debe truncarse en esta ventana."
    (and (stringp str)
         (not (string-blank-p str))
         (my/modeline--window-narrow-p)
         (> (length str) my/modeline-string-truncate-length)
         (not (one-window-p :no-minibuffer))))

  (defun my/modeline-string-cut-middle (str)
    "Recorta el medio de STR si procede, si no devuelve STR."
    (let ((half (floor my/modeline-string-truncate-length 2)))
      (if (my/modeline--string-truncate-p str)
          (concat (substring str 0 half) "..." (substring str (- half)))
        str)))

  (defun my/modeline--first-char (str)
    "Primer carácter de STR."
    (substring str 0 1))

  (defun my/modeline-string-abbreviate-but-last (str nthlast)
    "Abrevia STR por guiones/underscores dejando NTHLAST palabras intactas."
    (if (my/modeline--string-truncate-p str)
        (let* ((all (split-string str "[_-]"))
               (butlast-strings (nbutlast (copy-sequence all) nthlast))
               (last-strings (last (copy-sequence all) nthlast))
               (head (mapconcat #'my/modeline--first-char butlast-strings "-"))
               (tail (mapconcat #'identity last-strings "-")))
          (if (string-empty-p head) tail (concat head "-" tail)))
      str))

  ;; El color lo pone el tema en `my/modeline-set-faces'; aquí solo se declaran.
  (defface my/modeline-indicator-button nil
    "Base para los indicadores con fondo."
    :group 'mode-line)

  (dolist (color '("red" "green" "yellow" "blue" "magenta" "cyan" "gray"))
    (custom-declare-face (intern (format "my/modeline-indicator-%s" color))
                         '((t :inherit bold)) (format "Indicador %s." color)
                         :group 'mode-line)
    (custom-declare-face (intern (format "my/modeline-indicator-%s-bg" color))
                         '((t :inherit (bold my/modeline-indicator-button)))
                         (format "Indicador %s con fondo." color)
                         :group 'mode-line))
#+end_src

*** Segmentos
#+begin_src emacs-lisp
  (defvar-local my/modeline-narrow
      '(:eval
        (when (and (mode-line-window-selected-p)
                   (buffer-narrowed-p)
                   (not (memq major-mode '(Info-mode help-mode special-mode))))
          (propertize " Narrow " 'face 'my/modeline-indicator-cyan-bg)))
    "Indica que el buffer está estrechado.")

  (defvar-local my/modeline-buffer-status
      '(:eval
        (when (file-remote-p default-directory)
          (propertize " @ " 'face 'my/modeline-indicator-red-bg
                      'mouse-face 'mode-line-highlight)))
    "Indica que el archivo es remoto (TRAMP).")

  (defvar-local my/modeline-read-only
      '(:eval
        (when buffer-read-only
          (propertize " RO " 'face 'my/modeline-indicator-gray-bg
                      'mouse-face 'mode-line-highlight)))
    "Indica buffer de solo lectura.")

  (defun my/modeline-buffer-identification-face ()
    "Face para el nombre del buffer según foco y estado de modificación."
    (let ((modified (and (buffer-file-name) (buffer-modified-p))))
      (cond
       ((and (mode-line-window-selected-p) modified) '(italic mode-line-buffer-id))
       (modified 'italic)
       ((mode-line-window-selected-p) 'mode-line-buffer-id))))

  (defun my/modeline-buffer-name-help-echo ()
    "Tooltip con el nombre y la ruta del buffer."
    (concat (propertize (buffer-name) 'face 'mode-line-buffer-id) "\n"
            (propertize (or (buffer-file-name)
                            (format "Sin archivo.\nDirectorio: %s" default-directory))
                        'face 'font-lock-doc-face)))

  (defvar-local my/modeline-buffer-identification
      '(:eval
        (propertize (my/modeline-string-cut-middle (buffer-name))
                    'face (my/modeline-buffer-identification-face)
                    'mouse-face 'mode-line-highlight
                    'help-echo (my/modeline-buffer-name-help-echo)))
    "Nombre del buffer y su estado de modificación.")

  (defun my/modeline-major-mode-name ()
    "Nombre del `major-mode' capitalizado y sin el sufijo -mode."
    (capitalize (string-replace "-mode" "" (symbol-name major-mode))))

  (defun my/modeline-major-mode-help-echo ()
    "Tooltip con el símbolo del modo y su padre."
    (if-let* ((parent (get major-mode 'derived-mode-parent)))
        (format "Símbolo: `%s'.  Deriva de: `%s'" major-mode parent)
      (format "Símbolo: `%s'." major-mode)))

  (defvar-local my/modeline-major-mode
      (list
       (propertize "%[" 'face 'my/modeline-indicator-red)
       '(:eval

  (defun my/modeline-git--stash-count (root)
    "Número de stashes leyendo el reflog, para no lanzar un segundo proceso git."
    ;; ponytail: falla en worktrees (donde .git es un archivo); devuelve 0 y ya.
    (let ((f (expand-file-name ".git/logs/refs/stash" root)))
      (if (file-readable-p f)
          (with-temp-buffer
            (insert-file-contents f)
            (count-lines (point-min) (point-max)))

  (defun my/modeline-git--parse (output root)
    "Convierte OUTPUT de `git status --porcelain=v2 --branch' en un plist."
    (let ((branch nil) (ahead 0) (behind 0)
          (s-add 0) (s-mod 0) (s-del 0)
          (w-mod 0) (w-del 0) (untracked 0) (unmerged 0))
      (dolist (line (split-string output "\n" t))
        (cond
         ((string-prefix-p "# branch.head " line)
          (setq branch (substring line 14)))
         ((string-prefix-p "# branch.ab " line)
          (when (string-match "\\+\\([0-9]+\\) -\\([0-9]+\\)" line)
            (setq ahead (string-to-number (match-string 1 line))
                  behind (string-to-number (match-string 2 line)))))
         ((string-prefix-p "? " line) (cl-incf untracked))
         ((string-prefix-p "u " line) (cl-incf unmerged))
         ;; "1 XY ..." y "2 XY ..." : X = staged, Y = working tree
         ((and (> (length line) 3)
               (memq (aref line 0) '(?1 ?2))
               (eq (aref line 1) ?\s))
          (pcase (aref line 2)
            ((or ?A ?C) (cl-incf s-add))
            ((or ?M ?R) (cl-incf s-mod))
            (?D (cl-incf s-del)))
          (pcase (aref line 3)
            ((or ?M ?R) (cl-incf w-mod))
            (?D (cl-incf w-del))))))
      (list :branch branch :ahead ahead :behind behind
            :s-add s-add :s-mod s-mod :s-del s-del
            :w-mod w-mod :w-del w-del
            :untracked untracked :unmerged unmerged
            :stash (my/modeline-git--stash-count root))))

  (defun my/modeline-git-refresh (&rest _)
    "Recalcula el estado git del repo actual y refresca la mode line."
    (when-let* ((root (my/modeline-git--root)))
      (let* ((default-directory root)
             (out (with-temp-buffer
                    (when (zerop (call-process "git" nil t nil
                                               "status" "--porcelain=v2" "--branch"))
                      (buffer-string)))))
        (when out
          (puthash root (my/modeline-git--parse out root) my/modeline-git--cache)
          (force-mode-line-update t)))))

  ;; ponytail: git sincrono en hooks + timer inactivo. Si en un repo enorme se
  ;; nota el tiron, el upgrade es lanzarlo con `make-process' en asincrono.
  (dolist (h '(after-save-hook find-file-hook after-revert-hook))
    (add-hook h #'my/modeline-git-refresh))
  (with-eval-after-load 'magit
    (add-hook 'magit-post-refresh-hook #'my/modeline-git-refresh))
  (run-with-idle-timer 3 t #'my/modeline-git-refresh)

  (defun my/modeline-git--counts (pairs)
    "Formatea PAIRS ((símbolo . n)...) omitiendo los ceros, como oh-my-posh."
    (mapconcat (lambda (p) (format "%s%d" (car p) (cdr p)))
               (seq-filter (lambda (p) (> (cdr p) 0)) pairs)
               " "))

(defun my/org-add-block-overlays ()
  (save-excursion
    (let ((case-fold-search t))
      ;; Oculta todas las líneas #+
      (goto-char (point-min))
      (while (re-search-forward "^[ \t]*#\\+.*$" nil t)
        (let ((ov (make-overlay (match-beginning 0)
                                (min (1+ (match-end 0)) (point-max)))))
          (overlay-put ov 'invisible t)
          (push ov my/org-reading-overlays)))
      ;; Oculta filas de especificación de columna en tablas (<c>, <l>, etc.)
      (goto-char (point-min))
      (while (re-search-forward "^[ \t]*|\\([ \t]*<[lrcm]?[0-9]*>[ \t]*|\\)+[ \t]*$" nil t)
        (let ((ov (make-overlay (match-beginning 0)
                                (min (1+ (match-end 0)) (point-max)))))
          (overlay-put ov 'invisible t)
          (push ov my/org-reading-overlays)))
      ;; Estilo visual para bloques quote
      (goto-char (point-min))
      (while (re-search-forward "^[ \t]*#\\+begin_quote" nil t)
        (forward-line 1)
        (let ((start (point)))
          (when (re-search-forward "^[ \t]*#\\+end_quote" nil t)
            (let ((ov (make-overlay start (line-beginning-position))))
              (overlay-put ov 'face '(:slant italic))
              (overlay-put ov 'line-prefix "  │ ")
              (overlay-put ov 'wrap-prefix "    ")
              (push ov my/org-reading-overlays))))))))

  (use-package markdown-mode
    :ensure t
    :mode ("\\.md\\'" "\\.markdown\\'")
    :custom
    ;; Oculta los marcadores (* ** ` ~~) igual que org-hide-emphasis-markers
    (markdown-hide-markup nil)
    ;; Resalta código nativo con tree-sitter/font-lock según el lenguaje
    (markdown-fontify-code-blocks-natively t)
    ;; Escala los headers H1-H6 visualmente
    (markdown-header-scaling t)
    (markdown-header-scaling-values '(1.6 1.3 1.0 1.0 1.0 1.0))
    ;; Muestra un guión como viñeta en lugar de "-"
    (markdown-list-indent-width 2)
    :config
    (set-face-attribute 'markdown-header-face-1 nil :height 2.0)
    (set-face-attribute 'markdown-header-face-2 nil :height 1.6)
    (set-face-attribute 'markdown-header-face-3 nil :height 1.3)
    (set-face-attribute 'markdown-header-face-4 nil :height 1.0)
    (set-face-attribute 'markdown-header-face-5 nil :height 1.0)
    (set-face-attribute 'markdown-header-face-6 nil :height 1.0)
    (set-face-attribute 'markdown-code-face        nil :inherit 'fixed-pitch)
    (set-face-attribute 'markdown-pre-face         nil :inherit 'fixed-pitch)
    (set-face-attribute 'markdown-inline-code-face nil :inherit 'fixed-pitch)
    (set-face-attribute 'markdown-url-face         nil :inherit 'fixed-pitch)
    (add-hook 'markdown-mode-hook
              (lambda ()
                (font-lock-add-keywords nil
                  '(("^\\s-*\\(?:---\\|\\*\\*\\*\\|___\\)\\s-*$"
                     (0 '(face (:overline t) display (space :align-to right)) t))
                    (my/md-match-admonition (0 my/md-admonition-face prepend)))
                  t)))
    (defun my/markdown-markup-subtle ()
      (let* ((bg (face-attribute 'default :background))
             (fg (face-attribute 'default :foreground))
             (subtle (apply #'color-rgb-to-hex
                            (cl-mapcar (lambda (a b) (+ (* 0.70 a) (* 0.30 b)))
                                       (color-name-to-rgb bg)
                                       (color-name-to-rgb fg)))))
        (set-face-attribute 'markdown-markup-face nil :foreground subtle)))
    (with-eval-after-load 'ef-themes
      (add-hook 'ef-themes-post-load-hook #'my/markdown-markup-subtle))
    (my/markdown-markup-subtle)

    ;; Admoniciones estilo GitHub (> [!NOTE]) y RST (.. note::) en cajas de color.
    ;; note/tip → verde · warning/caution → amarillo · danger/important → rojo.
    (defface my/md-note    '((t :extend t)) "Caja verde: note/tip.")
    (defface my/md-warning '((t :extend t)) "Caja amarilla: warning/caution.")
    (defface my/md-danger  '((t :extend t)) "Caja roja: danger/important.")

    (defvar-local my/md-admonition-face nil
      "Face que `my/md-match-admonition' calcula para el bloque en curso.")

    (defun my/md-match-admonition (limit)
      "Matcher font-lock: cubre un bloque de admonición GitHub o RST hasta LIMIT.
Además oculta los marcadores (> y .. x::) con `invisible 'markdown-markup', que
markdown muestra/esconde con `markdown-toggle-markup-hiding' (modo lectura)."
      (when (re-search-forward
             (concat "^\\(?:"
                     ">[ \t]*\\[!\\(NOTE\\|TIP\\|WARNING\\|CAUTION\\|IMPORTANT\\|DANGER\\)\\]"
                     "\\|"
                     "\\.\\.[ \t]+\\(note\\|tip\\|warning\\|caution\\|important\\|danger\\)::"
                     "\\)")
             limit t)
        (let ((github (match-beginning 1))
              (kw (downcase (or (match-string 1) (match-string 2))))
              (mk-end (match-end 0))       ; fin del marcador [!X] o .. x::
              (beg (line-beginning-position))
              end)
          (setq my/md-admonition-face
                (pcase kw
                  ((or "note" "tip") 'my/md-note)
                  ((or "warning" "caution") 'my/md-warning)
                  (_ 'my/md-danger)))       ; important / danger
          (forward-line 1)
          (if github
              (while (and (< (point) limit) (looking-at "^>")) (forward-line 1))
            ;; RST: cuerpo indentado o en blanco hasta la 1ª línea sin indentar
            (while (and (< (point) limit)
                        (looking-at "^\\(?:[ \t]+\\S-\\|[ \t]*$\\)"))
              (forward-line 1)))
          (setq end (point))
          (with-silent-modifications
            (if github
                (save-excursion
                  ;; cabecera: si sólo lleva el marcador, colapsa la línea entera
                  (goto-char beg)
                  (put-text-property
                   beg (if (save-excursion (goto-char mk-end) (looking-at "[ \t]*$"))
                           (min (1+ (line-end-position)) end)
                         mk-end)
                   'invisible 'markdown-markup)
                  ;; cuerpo: oculta el "> " de cada línea
                  (forward-line 1)
                  (while (< (point) end)
                    (when (looking-at "^> ?")
                      (put-text-property (match-beginning 0) (match-end 0)
                                         'invisible 'markdown-markup))
                    (forward-line 1)))
              ;; RST: oculta el prefijo ".. x:: " y conserva el título
              (put-text-property
               beg (save-excursion (goto-char mk-end) (skip-chars-forward " \t") (point))
               'invisible 'markdown-markup)))
          (set-match-data (list beg end))
          t)))

    (defun my/md-admonition-colors ()
      "Tiñe las cajas mezclando el color base con el fondo del tema activo."
      (let ((bg (face-attribute 'default :background)))
        (cl-flet ((mix (color pct)
                    (apply #'color-rgb-to-hex
                           (cl-mapcar (lambda (a b) (+ (* pct a) (* (- 1.0 pct) b)))
                                      (color-name-to-rgb color)
                                      (color-name-to-rgb bg)))))
          (set-face-attribute 'my/md-note    nil :background (mix "#2f9e44" 0.18))
          (set-face-attribute 'my/md-warning nil :background (mix "#e6a700" 0.20))
          (set-face-attribute 'my/md-danger  nil :background (mix "#e03131" 0.18)))))
    (with-eval-after-load 'ef-themes
      (add-hook 'ef-themes-post-load-hook #'my/md-admonition-colors))
    (my/md-admonition-colors)

    (defun my/md-admonition-selftest ()
      "Comprobación mínima del matcher. Ejecuta con \\[execute-extended-command]."
      (interactive)
      (with-temp-buffer
        (insert "> [!WARNING]\n> ojo\n\ntras\n")
        (goto-char (point-min))
        (cl-assert (my/md-match-admonition (point-max)))
        (cl-assert (eq my/md-admonition-face 'my/md-warning))
        ;; el "> " de la cabecera y del cuerpo quedan marcados invisibles
        (cl-assert (eq (get-text-property (point-min) 'invisible) 'markdown-markup))
        (cl-assert (eq (get-text-property (- (point-max) 3) 'invisible) 'markdown-markup))
        (erase-buffer)
        (insert ".. note:: hola\n   cuerpo\n\nfuera\n")
        (goto-char (point-min))
        (cl-assert (my/md-match-admonition (point-max)))
        (cl-assert (eq my/md-admonition-face 'my/md-note))
        ;; ".. note:: " invisible, el título "hola" visible
        (cl-assert (eq (get-text-property (point-min) 'invisible) 'markdown-markup))
        (cl-assert (null (get-text-property (- (line-end-position) 1) 'invisible))))
      (message "admonition selftest OK"))
    (defun my/markdown-sync-reading-style (&rest _)
